building Theia¶

apt-get install libgflags2 libgflags-dev
apt-get install freeglut3 freeglut3-dev
apt-get install libxmu-dev libxi-dev

git clone https://github.com/sweeneychris/TheiaSfM.git
cd TheiaSfM/
mkdir theia-build
cd theia-build
cmake ..
make -j4
